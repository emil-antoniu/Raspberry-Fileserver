# Setup for image backups.
sudo mkdir /media/usb
# Mount external drive onto which the image will be copied. Double-check partition name.
sudo mount /dev/sda1 /media/usbsu
sudo su
sudo dd if=/dev/mmcblk0 bs=4M | gzip > /media/usb/arc-fileserver-backup.img.gz

# Startup.
## Mounting external storage.
lsblk -o NAME,UUID,MOUNTPOINT,FSTYPE
sudo nano /etc/fstab
UUID=XXXXX /media/hdd ntfs-3g defaults,noatime,bigwrites 0 2
sudo mount -a
## Starting qbittorrent-nox on boot.
sudo nano /etc/systemd/system/qbittorrent-nox.service
[Unit]
Description=qBittorrent-nox
After=network.target

[Service]
User=pi
ExecStart=/usr/bin/qbittorrent-nox
Restart=always

[Install]
WantedBy=multi-user.target

sudo systemctl enable qbittorrent-nox  
sudo systemctl start qbittorrent-nox

# UFW configurations.
sudo ufw allow ssh
sudo ufw default deny incoming
# Allow 8080 for qbittorrent-nox WebUI.
sudo ufw allow 8080/tcp
# Allow qbittorrent-nox to listen on 57683.
sudo ufw allow 57683/tcp  
sudo ufw allow 57683/udp  
# Allow qbittorrent-nox to use default peer communication ports.
sudo ufw allow 6881:6889/tcp  
sudo ufw allow 6881:6889/udp 
# Allow Samba to distribute files on LAN.
sudo ufw allow Samba
sudo ufw enable

# qbittorrent-nox configurations.

## WebUI
### Change default username and password.
### Use the interface with the VPN only (generally tun0).
### Change default save path to /media/hdd.
### Disable UPnP.
### Automatically add torrents from /media/hdd.

# Samba configurations.
sudo nano /etc/samba/smb.conf
## smb.conf file should only contain the following.
[global]
   workgroup = WORKGROUP
   server string = Raspberry Pi Samba Server
   security = user
   map to guest = Bad User

[hdd]
   comment = MAIN STORAGE UNIT
   path = /media/hdd
   browseable = yes
   writable = yes
   create mask = 0777
   directory mask = 0777
   guest ok = yes
   public = yes
   force user = pi
sudo systemctl restart smbd
sudo systemctl status smbd

# External storage configuations.
sudo mkdir /media/hdd
# Double-check partition name.
sudo mount /dev/sda1 /media/hdd
